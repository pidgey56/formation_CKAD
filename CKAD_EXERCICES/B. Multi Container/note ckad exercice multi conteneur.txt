CKAD _ exercice:
    Multi Container:

        1: 
            Create a pod with two containers, both with image busybox and command "echo hello; sleep 3600". Connect to the second container and run 'ls'
                Action: 
                    k run busyb --image=busybox --restart=Never -o yaml --dry-run=client --command -- /bin/sh -c "echo 'helloworld'; sleep 300" > exo.yaml
                    Editer le fichier exo.yaml pour ajouter un second container. 
                    kubectl apply -f exo.yaml
                    kubectl exec -it busyb -c busyb-2 -- ls #ou /bin/sh et executer sur le conteneur la commande ls

        2:
            Create a pod with an nginx container exposed on port 80. 
            Add a busybox init container which downloads a page 
            using 'echo "test" > /work-dir/index.html'. 
            Make a volume of type emptyDir and mount it in both containers. 
            For the nginx container, mount it on "/usr/share/nginx/html" 
            and for the initcontainer, mount it on "/work-dir". 
            When done, get the IP of the created pod and create a 
            busybox pod and run "wget -O- IP"
                Action:
                    k run nginx --image=nginx --restart=Never -o yaml --dry-run=client > exo2.yaml
                    # Apply pod
                    kubectl apply -f exo2.yaml

                    k run busybox --image=busybox --restart=Never -o yaml --dry-run=client > bud_exo2.yaml
                    # Get IP
                    kubectl get po -o wide
                    # get pod IP of nginx pod 
                    Editer le fichier bud_exo2.yaml pour ajouter la commande wget vers l'IP du pod nginx
                    # Execute wget
                    kubectl apply -f bud_exo2.yaml
                    k logs busybox
